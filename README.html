BayesRates

Using Bayesian methods, project demographic indicators. Optionally, incorporate expert judgment into the projections by specifying benchmarks.

BayesProj is part of a larger project to extend the ProFamy household projection method.

Work on BayesRates has been supported by funding from the Chinese Ministry of Science and Technology.

Installation

Install the development version of BayesProj from GitHub with:

# install.packages("devtools")
devtools::install_github("junnizhang/BayesProj")

Example

library(BayesRates)
library(BayesProj)
library(dplyr)
library(ggplot2)

Smooth Chinese data on divorce rates

res <- smooth_agetime(nevent_df = cn_divorces,
                      py_df = cn_population,
                      spec_time = TimeFixed(),
                      byvar = "sex")

Calculate the total divorce rate

TDR <- total_rate(res)

Extract results

data <- TDR |>
  select(sex, time, TDR = .probability)

Fit a time series model to the historical estimates

fitted <- fit_ts(data, 
                 indvar = "TDR",
                 byvar = "sex")

Project future values

projected <- project_ts(fitted, 
                        time_labels = 2019:2028)

Extract historical and projected values and plot them

vals <- composite(projected)

ggplot(vals, aes(x = time, y = TDR)) +
  facet_wrap(vars(sex)) +
  geom_ribbon(aes(ymin = .lower, ymax = .upper), 
              fill = "salmon") +
  geom_line(col = "darkred", 
            linewidth = 0.5)

Create benchmarks

bench <- Benchmarks(data.frame(time = c(2028, 2028),
                               sex = c("Female", "Male"),
                               q50 = c(0.15, 0.25),
                               q90 = c(0.2, 0.3)))

Re-do the projection, with benchmarks

projected_bench <- project_ts(fitted, 
                              time_labels = 2019:2028,
                  spec_bench = bench)

vals_bench <- composite(projected_bench)

ggplot(vals_bench, aes(x = time, y = TDR)) +
  facet_wrap(vars(sex)) +
  geom_ribbon(aes(ymin = .lower, ymax = .upper), 
              fill = "salmon") +
  geom_line(col = "darkred", 
            linewidth = 0.5)
